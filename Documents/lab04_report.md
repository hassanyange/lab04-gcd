def extended_euclidean_algorithm(a, b):
    r0, r1 = a, b
    x0, x1 = 1, 0
    y0, y1 = 0, 1
    
    while r1 != 0:
        qi = r0 // r1
        r0, r1 = r1, r0 % r1
        x0, x1 = x1, x0 - qi * x1
        y0, y1 = y1, y0 - qi * y1
    
    return r0, x0, y0
